# 1、常用命令：

```bash
sudo apt install htop

htop
	-h                     # 查看帮助
	-d <num>               # 更新间隔(秒)
```

# 2、常见用法：

```bash
# 交互式命令
q/F10           # 退出
P               # 按cpu占用排序
M               # 按内存占用排序
T               # 按运行时间排序
t/F5            # 树形显示
p               # 显示命令绝对路径
u               # 选择显示指定用户的进程
H               # 不显示用户进程
K               # 不显示内核进程
上下键           # 选择进程
    e               # 显示进程环境变量
    l               # 显示打开的文件
```
