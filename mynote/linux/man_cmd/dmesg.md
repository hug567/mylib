# 1、常用命令：

```bash
dmesg
	-w                      # 实时输出日志
	-C                      # 清除内核日志buffer
	-c                      # 读取内核日志后清除buffer
	-n <level>              # 设置日志打印级别
	-r                      # 打印原始日志(头部有日志级别)
	-k                      # 只打印内核日志
	-u                      # 只打印用户态日志
	-x                      # 显示日志级别
	-d                      # 显示日志间的时间间隔
	-e                      # 显示真实时间和日志时间间隔
	-t                      # 不显示时间戳
	-T                      # 已可读形式显示时间
```

# 2、常见用法：

```bash
# 打印日志
dmesg
# 实时打印日志
dmesg -w
# 打印日志级别和时间间隔
dmesg -xd
```
