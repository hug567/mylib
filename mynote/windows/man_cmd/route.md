# 1、命令参数：

```bash
route
```

# 2、常见用法：

```bash
# 查看路由表
route print
# 查看ipv4路由器表
route print -4

# 添加路由：需以管理员身份打开cmd
route add 172.22.195.0 mask 255.255.255.0 172.22.192.1
# 添加默认路由
route add 0.0.0.0 mask 0.0.0.0 192.168.99.106
# 添加路由：192.168.99.0/24网段的网络包，从192.168.98.51的网卡收发
route add 192.168.99.0 mask 255.255.255.9 192.168.98.51

# 删除路由：172.22.195.0为第一列的网络目标
route delete 172.22.195.0
# 修改路由
route change ...
```
